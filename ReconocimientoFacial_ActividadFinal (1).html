Abre detector.py y escribe el siguiente código

pythondetector.py — Captura básica de cámara
import cv2

def iniciar_camara(indice=0):
    """Abre la cámara y devuelve el objeto VideoCapture."""
    cap = cv2.VideoCapture(indice)
    if not cap.isOpened():
        raise RuntimeError("No se pudo abrir la cámara. Verifica el índice.")
    # Configurar resolución
    cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
    cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
    return cap

def mostrar_camara():
    """Muestra el feed de la cámara en tiempo real. Presiona 'q' para salir."""
    cap = iniciar_camara()

    while True:
        ret, frame = cap.read()
        if not ret:
            break

        # Voltear horizontalmente (efecto espejo, más natural para el usuario)
        frame = cv2.flip(frame, 1)

        # Mostrar el frame en una ventana
        cv2.imshow("Cámara — Presiona Q para salir", frame)

        # Salir si se presiona 'q'
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

    cap.release()
    cv2.destroyAllWindows()

# Prueba directa del módulo
if __name__ == "__main__":
    mostrar_camara()

Detección de Rostros en Tiempo Real

Usarás la librería face_recognition para localizar rostros en cada frame.

🔍
¿Cómo funciona la detección facial? face_recognition usa un modelo HOG (Histogram of Oriented Gradients) para detección rápida, o una CNN para mayor precisión. Devuelve las coordenadas de cada rostro detectado como una tupla (top, right, bottom, left).

Agrega estas funciones a detector.py

pythondetector.py — Detección de rostros
import cv2
import face_recognition

def detectar_rostros(frame, modelo="hog"):
    """
    Detecta ubicaciones de rostros en un frame.
    modelo: 'hog' (rápido) o 'cnn' (más preciso, requiere GPU)
    Retorna: lista de tuplas (top, right, bottom, left)
    """
    # face_recognition trabaja con RGB; OpenCV usa BGR
    rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)

    # Reducir el frame para acelerar la detección (procesa cada 2 píxeles)
    rgb_pequeño = cv2.resize(rgb, (0, 0), fx=0.5, fy=0.5)

    ubicaciones = face_recognition.face_locations(rgb_pequeño, model=modelo)

    # Escalar de vuelta las coordenadas al tamaño original
    ubicaciones_reales = [
        (top * 2, right * 2, bottom * 2, left * 2)
        for top, right, bottom, left in ubicaciones
    ]
    return ubicaciones_reales

def dibujar_rostros(frame, ubicaciones, etiquetas=None):
    """
    Dibuja rectángulos y etiquetas sobre los rostros detectados.
    etiquetas: lista de strings con los nombres. Si es None, muestra "Detectado".
    """
    for i, (top, right, bottom, left) in enumerate(ubicaciones):
        # Color del rectángulo
        color = (0, 255, 0)  # Verde (BGR)

        # Rectángulo alrededor del rostro
        cv2.rectangle(frame, (left, top), (right, bottom), color, 2)

        # Etiqueta debajo del rectángulo
        etiqueta = etiquetas[i] if etiquetas else "Detectado"
        cv2.rectangle(frame, (left, bottom - 28), (right, bottom), color, cv2.FILLED)
        cv2.putText(
            frame, etiqueta,
            (left + 6, bottom - 8),
            cv2.FONT_HERSHEY_DUPLEX, 0.6,
            (0, 0, 0), 1
        )
    return frame

def demo_deteccion():
    """Muestra la cámara con detección de rostros en tiempo real."""
    cap = iniciar_camara()

    while True:
        ret, frame = cap.read()
        if not ret:
            break

        frame = cv2.flip(frame, 1)
        ubicaciones = detectar_rostros(frame)
        frame = dibujar_rostros(frame, ubicaciones)

        # Mostrar cantidad de rostros detectados
        texto = f"Rostros detectados: {len(ubicaciones)}"
        cv2.putText(frame, texto, (10, 30),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.8, (0, 255, 255), 2)

        cv2.imshow("Detección Facial — Q para salir", frame)
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

    cap.release()
    cv2.destroyAllWindows()

if __name__ == "__main__":
    demo_deteccion()

Crea el archivo utils.py con herramientas de persistencia

pythonutils.py — Guardar y cargar la base de datos
import pickle
import os

DB_PATH = "database/usuarios.pkl"

def cargar_base_datos():
    """Carga la base de datos. Devuelve dict vacío si no existe."""
    if os.path.exists(DB_PATH) and os.path.getsize(DB_PATH) > 0:
        with open(DB_PATH, "rb") as f:
            return pickle.load(f)
    return {}  # {nombre: [embedding1, embedding2, ...]}

def guardar_base_datos(db):
    """Guarda el diccionario de usuarios en disco."""
    os.makedirs("database", exist_ok=True)
    with open(DB_PATH, "wb") as f:
        pickle.dump(db, f)
    print(f"Base de datos guardada: {len(db)} usuario(s).")

def listar_usuarios():
    """Imprime todos los usuarios registrados."""
    db = cargar_base_datos()
    if not db:
        print("No hay usuarios registrados.")
        return
    print(f"\n{'='*30}")
    print(f"  USUARIOS REGISTRADOS ({len(db)})")
    print(f"{'='*30}")
    for nombre, embeddings in db.items():
        print(f"  • {nombre}: {len(embeddings)} muestra(s)")
    print(f"{'='*30}\n")

if __name__ == "__main__":
    listar_usuarios()

Crea el archivo registrar.py

pythonregistrar.py — Registro de nuevos usuarios
import cv2
import face_recognition
import os
from detector import iniciar_camara, detectar_rostros, dibujar_rostros
from utils import cargar_base_datos, guardar_base_datos

FOTOS_DIR = "database/fotos"

def capturar_embedding(frame, ubicacion):
    """
    Extrae el embedding (vector de 128 dimensiones) de un rostro en el frame.
    ubicacion: tupla (top, right, bottom, left) del rostro.
    """
    rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
    # face_encodings necesita la lista de ubicaciones
    encodings = face_recognition.face_encodings(rgb, [ubicacion])
    if encodings:
        return encodings[0]  # Vector NumPy de 128 floats
    return None

def guardar_foto_usuario(frame, nombre, muestra_num):
    """Guarda la imagen del registro en disco."""
    carpeta = os.path.join(FOTOS_DIR, nombre)
    os.makedirs(carpeta, exist_ok=True)
    ruta = os.path.join(carpeta, f"muestra_{muestra_num}.jpg")
    cv2.imwrite(ruta, frame)
    print(f"Foto guardada: {ruta}")

def registrar_usuario():
    """
    Flujo de registro interactivo:
    1. Detecta un rostro en la cámara
    2. Pide el nombre del usuario
    3. Captura N muestras para mayor precisión
    4. Guarda embeddings en la base de datos
    """
    cap = iniciar_camara()
    db = cargar_base_datos()

    nombre = input("\nIngresa el nombre del nuevo usuario: ").strip()
    if not nombre:
        print("Nombre inválido.")
        return

    if nombre in db:
        resp = input(f"'{nombre}' ya existe. ¿Agregar más muestras? (s/n): ")
        if resp.lower() != 's':
            return

    MUESTRAS_REQUERIDAS = 5
    muestras_capturadas = 0
    nuevos_embeddings = []

    print(f"\nRegistrando a '{nombre}'. Necesitamos {MUESTRAS_REQUERIDAS} muestras.")
    print("Presiona ESPACIO cuando tengas el rostro bien encuadrado.")
    print("Presiona Q para cancelar.\n")

    while muestras_capturadas < MUESTRAS_REQUERIDAS:
        ret, frame = cap.read()
        if not ret:
            break

        frame = cv2.flip(frame, 1)
        ubicaciones = detectar_rostros(frame)
        frame_display = dibujar_rostros(frame.copy(), ubicaciones)

        # HUD informativo
        info = f"Muestras: {muestras_capturadas}/{MUESTRAS_REQUERIDAS} | ESPACIO=capturar Q=cancelar"
        cv2.putText(frame_display, info, (10, 25),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.55, (255, 255, 0), 1)

        cv2.imshow(f"Registrando: {nombre}", frame_display)
        tecla = cv2.waitKey(1) & 0xFF

        if tecla == ord('q'):
            print("Registro cancelado.")
            break

        if tecla == ord(' ') and ubicaciones:
            ubicacion = ubicaciones[0]
            embedding = capturar_embedding(frame, ubicacion)

            if embedding is not None:
                nuevos_embeddings.append(embedding)
                guardar_foto_usuario(frame, nombre, muestras_capturadas + 1)
                muestras_capturadas += 1
                print(f"  ✓ Muestra {muestras_capturadas} capturada.")
            else:
                print("  ✗ No se pudo extraer el embedding. Inténtalo de nuevo.")

        elif tecla == ord(' ') and not ubicaciones:
            print("  ✗ No hay rostro detectado en el frame.")

    cap.release()
    cv2.destroyAllWindows()

    # Guardar si se obtuvieron muestras
    if nuevos_embeddings:
        if nombre not in db:
            db[nombre] = []
        db[nombre].extend(nuevos_embeddings)
        guardar_base_datos(db)
        print(f"\n✅ '{nombre}' registrado con {len(nuevos_embeddings)} muestra(s).")

if __name__ == "__main__":
    registrar_usuario()

Persistencia con SQLite — Reemplazando Pickle

En esta sección migrarás el almacenamiento de pickle a una base de datos relacional con SQLite. Diseñarás el esquema, escribirás las consultas SQL y crearás el módulo de acceso a datos que usará todo el sistema a partir de ahora.

S.1

¿Por qué Base de Datos Relacional?

Antes de escribir código, entiende por qué pickle no es suficiente en un sistema real.

📦
Limitaciones de pickle El archivo usuarios.pkl que construiste en la Fase 1 funciona, pero tiene problemas serios:
  • No soporta consultas — para buscar un usuario debes cargar todos los datos en memoria.
  • No tiene control de concurrencia — si dos procesos escriben al mismo tiempo, el archivo se corrompe.
  • No hay historial, auditoría, ni relaciones entre tablas.
  • No es portable — un archivo .pkl generado en Python 3.8 puede no abrirse en 3.11.
🗃️
Ventajas de SQLite
  • Base de datos relacional completa en un solo archivo .db.
  • Soporte nativo en Python con el módulo sqlite3 — sin instalación extra.
  • Permite consultas SQL estándar: SELECT, INSERT, JOIN, WHERE.
  • Transacciones ACID: si algo falla a la mitad, la base de datos queda íntegra.
  • Fácilmente escalable a PostgreSQL o MySQL en producción cambiando solo el conector.

import sqlite3
import numpy as np
import os
from datetime import datetime

DB_PATH = "database/reconocimiento.db"

# ─────────────────────────────────────────
#  CONEXIÓN
# ─────────────────────────────────────────
def obtener_conexion():
    """
    Devuelve una conexión a la base de datos SQLite.
    Activa las llaves foráneas y el modo WAL para mejor concurrencia.
    """
    os.makedirs("database", exist_ok=True)
    conn = sqlite3.connect(DB_PATH)
    conn.execute("PRAGMA foreign_keys = ON")

def crear_tablas():
    """
    Crea las tablas del sistema si no existen.
    Seguro ejecutarlo múltiples veces (usa IF NOT EXISTS).
    """
    conn = obtener_conexion()
    cursor = conn.cursor()

    # Tabla principal de usuarios
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS usuarios (
            id          INTEGER PRIMARY KEY AUTOINCREMENT,
            nombre      TEXT    NOT NULL,
            apellido    TEXT    DEFAULT '',
            email       TEXT    DEFAULT '',
            fecha_reg   TEXT    NOT NULL,
            activo      INTEGER DEFAULT 1
        )
    """)

    # Tabla de embeddings faciales (N muestras por usuario)
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS embeddings (
            id          INTEGER PRIMARY KEY AUTOINCREMENT,
            usuario_id  INTEGER NOT NULL,
            embedding   BLOB    NOT NULL,
            muestra_num INTEGER NOT NULL,
            fecha_cap   TEXT    NOT NULL,
            FOREIGN KEY (usuario_id) REFERENCES usuarios(id)
                ON DELETE CASCADE
        )
    """)

    # Tabla de historial de accesos
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS accesos (
            id          INTEGER PRIMARY KEY AUTOINCREMENT,
            usuario_id  INTEGER,
            timestamp   TEXT    NOT NULL,
            confianza   REAL    DEFAULT 0.0,
            estado      TEXT    NOT NULL,
            camara_id   TEXT    DEFAULT 'cam_0',
            FOREIGN KEY (usuario_id) REFERENCES usuarios(id)
                ON DELETE SET NULL
        )
    """)

    # Índice para acelerar búsquedas por usuario_id en embeddings
    cursor.execute("""
        CREATE INDEX IF NOT EXISTS idx_embeddings_usuario
        ON embeddings(usuario_id)
    """)

    conn.commit()
    conn.close()
    print("✅ Base de datos inicializada correctamente.")

def embedding_a_blob(embedding_np):
    """Convierte array NumPy (128 floats) a bytes para guardar en BLOB."""
    return embedding_np.astype(np.float64).tobytes()

def blob_a_embedding(blob):
    """Convierte bytes BLOB de vuelta a array NumPy de 128 floats."""
    return np.frombuffer(blob, dtype=np.float64)

    Registra un evento de reconocimiento en la tabla accesos.
    Busca el usuario_id por nombre para mantener la integridad referencial.

    confianza = max(0.0, (1 - distancia / umbral) * 100)

    usuario_id = None
    if nombre_detectado != "Desconocido":
        cursor.execute(
            "SELECT id FROM usuarios WHERE nombre = ? AND activo = 1",

        if fila:
            usuario_id = fila['id']

    cursor.execute(
        """INSERT INTO accesos (usuario_id, timestamp, confianza, estado, camara_id)
           VALUES (?, ?, ?, ?, ?)""",
        (usuario_id, datetime.now().isoformat(), round(confianza, 2), estado, camara_id)
    )
    conn.commit()
    conn.close()

def capturar_embedding(frame, ubicacion):
    """Extrae el vector de 128 dimensiones de un rostro en el frame."""
    rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
    encodings = face_recognition.face_encodings(rgb, [ubicacion])

def guardar_foto_usuario(frame, nombre, muestra_num):
    """Guarda la foto de la captura en disco como respaldo visual."""
    carpeta = os.path.join(FOTOS_DIR, nombre)
    os.makedirs(carpeta, exist_ok=True)
    ruta = os.path.join(carpeta, f"muestra_{muestra_num}.jpg")

def registrar_usuario():
    """
    Flujo de registro con persistencia en SQLite:
    1. Solicita nombre, apellido y email del nuevo usuario
    2. Inserta en tabla 'usuarios' y obtiene el ID asignado
    3. Captura N muestras faciales
    4. Inserta cada embedding en tabla 'embeddings' con el usuario_id
    """
    crear_tablas()  # Asegura que las tablas existan
    cap = iniciar_camara()

    print("\n── REGISTRO DE NUEVO USUARIO ──")
    nombre   = input("Nombre:   ").strip()
    apellido = input("Apellido: ").strip()
    email    = input("Email:    ").strip()

    if not nombre:
        print("El nombre es obligatorio.")
        cap.release()
        return

    # Insertar en BD y obtener ID (si ya existe retorna el ID existente)
    usuario_id = insertar_usuario(nombre, apellido, email)

    # Cuántas muestras ya tiene (por si se agregan más)
    muestras_previas  = contar_embeddings_usuario(usuario_id)
    MUESTRAS_NUEVAS   = 5
    capturadas        = 0

    print(f"\nRegistrando a '{nombre}' (ID={usuario_id})")
    print(f"Muestras previas: {muestras_previas} | Nuevas a capturar: {MUESTRAS_NUEVAS}")
    print("Presiona ESPACIO para capturar, Q para cancelar.\n")

    while capturadas < MUESTRAS_NUEVAS:
        ret, frame = cap.read()
        if not ret:
            break
        frame = cv2.flip(frame, 1)
        ubicaciones = detectar_rostros(frame)
        display = dibujar_rostros(frame.copy(), ubicaciones)

        info = f"Capturadas: {capturadas}/{MUESTRAS_NUEVAS} | ESPACIO=capturar  Q=cancelar"
        cv2.putText(display, info, (10, 25),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.55, (255, 255, 0), 1)
        cv2.imshow(f"Registrando: {nombre} | ID {usuario_id}", display)

        tecla = cv2.waitKey(1) & 0xFF
        if tecla == ord('q'):
            break
        if tecla == ord(' ') and ubicaciones:
            embedding = capturar_embedding(frame, ubicaciones[0])
            if embedding is not None:
                num_muestra = muestras_previas + capturadas + 1
                # INSERT en tabla embeddings con el usuario_id como FK
                insertar_embedding(usuario_id, embedding, num_muestra)
                guardar_foto_usuario(frame, nombre, num_muestra)
                capturadas += 1
                print(f"  ✓ Muestra {num_muestra} guardada en BD (usuario_id={usuario_id})")
            else:
                print("  ✗ No se pudo extraer el embedding.")

    cap.release()
    cv2.destroyAllWindows()

    total_final = contar_embeddings_usuario(usuario_id)
    print(f"\n✅ Registro completado. '{nombre}' tiene ahora {total_final} muestra(s) en BD.")

Motor de Reconocimiento con Umbral de Similitud

Aprenderás a comparar embeddings usando distancia euclidiana y a definir un umbral de confianza.

📐
Concepto clave: Distancia entre embeddings Para comparar dos rostros calculamos la distancia euclidiana entre sus vectores de 128 dimensiones. Si la distancia es menor a un umbral (típicamente 0.6), los rostros pertenecen a la misma persona. Menor distancia = mayor similitud.
⚠️
El umbral es crítico Un umbral muy bajo genera muchos falsos negativos (no reconoce al usuario aunque esté registrado). Muy alto genera falsos positivos (confunde personas). El valor por defecto de face_recognition es 0.6. Lo ajustaremos experimentalmente.

pythonreconocer.py — Motor de reconocimiento facial
import cv2
import face_recognition
import numpy as np
from detector import iniciar_camara, detectar_rostros, dibujar_rostros
from utils import cargar_base_datos
from registrar import capturar_embedding

UMBRAL_RECONOCIMIENTO = 0.55  # Ajusta este valor según tus pruebas

def identificar_rostro(embedding_desconocido, db):
    """
    Compara un embedding desconocido contra toda la base de datos.
    
    Retorna: (nombre, distancia_minima)
    Si la distancia mínima supera el umbral, retorna ("Desconocido", distancia)
    """
    mejor_nombre = "Desconocido"
    mejor_distancia = float("inf")

    for nombre, embeddings in db.items():
        # Calcular distancias contra todas las muestras de este usuario
        distancias = face_recognition.face_distance(embeddings, embedding_desconocido)
        
        # Usar la distancia mínima (mejor match) para este usuario
        dist_min = np.min(distancias)

        if dist_min < mejor_distancia:
            mejor_distancia = dist_min
            mejor_nombre = nombre

    # Aplicar umbral
    if mejor_distancia > UMBRAL_RECONOCIMIENTO:
        return "Desconocido", mejor_distancia

    return mejor_nombre, mejor_distancia

def procesar_frame_reconocimiento(frame, db):
    """
    Detecta y reconoce todos los rostros en un frame.
    Retorna: frame anotado, lista de (nombre, distancia, ubicacion)
    """
    ubicaciones = detectar_rostros(frame)
    resultados = []
    etiquetas = []

    if not ubicaciones:
        return frame, []

    rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)

    # Obtener embeddings para todas las ubicaciones de una vez (más eficiente)
    rgb_pequeño = cv2.resize(rgb, (0, 0), fx=0.5, fy=0.5)
    ubicaciones_pequeñas = [(t//2, r//2, b//2, l//2) for t, r, b, l in ubicaciones]
    encodings = face_recognition.face_encodings(rgb_pequeño, ubicaciones_pequeñas)

    for i, embedding in enumerate(encodings):
        nombre, distancia = identificar_rostro(embedding, db)
        confianza = max(0, (1 - distancia / UMBRAL_RECONOCIMIENTO) * 100)
        
        etiqueta = f"{nombre} ({confianza:.0f}%)" if nombre != "Desconocido" else "Desconocido"
        etiquetas.append(etiqueta)
        resultados.append((nombre, distancia, ubicaciones[i]))

    frame = dibujar_rostros(frame, ubicaciones, etiquetas)
    return frame, resultados

def evaluar_con_fotos_guardadas():
    """
    Evalúa el sistema usando las fotos de registro como conjunto de prueba.
    Calcula precisión, matriz de confusión simplificada y errores por usuario.

    db = cargar_base_datos()

    total = 0
    correctos = 0

    for nombre_real in os.listdir("database/fotos"):
        carpeta = os.path.join("database/fotos", nombre_real)
        fotos = [f for f in os.listdir(carpeta) if f.endswith(".jpg")]

        aciertos_usuario = 0
        for foto in fotos:
            ruta = os.path.join(carpeta, foto)
            img = face_recognition.load_image_file(ruta)
            encodings = face_recognition.face_encodings(img)

            if not encodings:
                continue

            nombre_pred, dist = identificar_rostro(encodings[0], db)
            total += 1

            if nombre_pred == nombre_real:
                correctos += 1
                aciertos_usuario += 1
            else:
                errores.append((nombre_real, nombre_pred, dist, foto))

        prec_usuario = (aciertos_usuario / len(fotos) * 100) if fotos else 0
        print(f"  {nombre_real}: {aciertos_usuario}/{len(fotos)} ({prec_usuario:.0f}%)")

    precision_global = (correctos / total * 100) if total > 0 else 0
    print(f"\n{'='*40}")
    print(f"  Total evaluado:  {total} muestras")
    print(f"  Correctos:       {correctos}")
    print(f"  Errores:         {total - correctos}")
    print(f"  Precisión global: {precision_global:.1f}%")
    print(f"  Umbral usado:    {UMBRAL_RECONOCIMIENTO}")
    print(f"{'='*40}\n")

    if errores:
        print("Errores de clasificación:")
        for real, pred, dist, foto in errores:
            print(f"  Foto: {foto} | Real: {real} → Predicho: {pred} (dist={dist:.3f})")

if __name__ == "__main__":

Busca el paper de Soukupová & Čech (2016): "Real-Time Eye Blink Detection using Facial Landmarks". El EAR se calcula con las distancias verticales y horizontal del ojo. Cuando parpadeas, las distancias verticales se reducen a ~0.

import dlib
import numpy as np
from scipy.spatial import distance as dist

# Índices de landmarks para ojos izquierdo y derecho (modelo 68 puntos)
OJO_IZQ = (42, 48)  # índice inicio y fin en los landmarks
OJO_DER = (36, 42)

EAR_UMBRAL = 0.25  # Debajo de este valor se considera parpadeo
FRAMES_PARPADEO = 2   # Frames consecutivos para confirmar parpadeo

predictor = dlib.shape_predictor("shape_predictor_68_face_landmarks.dat")
detector_dlib = dlib.get_frontal_face_detector()

def calcular_ear(ojo_pts):
    """Calcula el Eye Aspect Ratio para un conjunto de 6 puntos del ojo."""
    A = dist.euclidean(ojo_pts[1], ojo_pts[5])
    B = dist.euclidean(ojo_pts[2], ojo_pts[4])
    C = dist.euclidean(ojo_pts[0], ojo_pts[3])
    return (A + B) / (2.0 * C)

class DetectorVivacidad:
    def __init__(self):
        self.contador_frames_bajo = 0
        self.total_parpadeos = 0
        self.es_vivo = False
        self.PARPADEOS_REQUERIDOS = 2  # Debe parpadear al menos 2 veces

    def procesar(self, frame_gray, rect_dlib):
        """
        Analiza un frame en escala de grises para detectar parpadeo.
        rect_dlib: rectángulo de dlib del rostro detectado.
        Retorna: (ear_promedio, es_vivo)
        """
        shape = predictor(frame_gray, rect_dlib)
        pts = np.array([[shape.part(i).x, shape.part(i).y] 
                        for i in range(68)])

        ojo_izq_pts = pts[OJO_IZQ[0]:OJO_IZQ[1]]
        ojo_der_pts = pts[OJO_DER[0]:OJO_DER[1]]

        ear_izq = calcular_ear(ojo_izq_pts)
        ear_der = calcular_ear(ojo_der_pts)
        ear = (ear_izq + ear_der) / 2.0

        if ear < EAR_UMBRAL:
            self.contador_frames_bajo += 1
        else:
            if self.contador_frames_bajo >= FRAMES_PARPADEO:
                self.total_parpadeos += 1
                if self.total_parpadeos >= self.PARPADEOS_REQUERIDOS:
                    self.es_vivo = True
            self.contador_frames_bajo = 0

        return ear, self.es_vivo
